Redis客户端之Go客户端：redigo简介

一、获取redigo

go get github.com/garyburd/redigo/redis

二、简单使用

c, err := redis.Dial("tcp", "127.0.0.1:6379")
if err != nil {
    fmt.Println(err)
    return
}
defer c.Close()
v, err := c.Do("SET", "hello", "world")
if err != nil {
    fmt.Println(err)
    return
}
fmt.Println(v)
v, err = redis.String(c.Do("GET", "hello"))
if err != nil {
    fmt.Println(err)
    return
}
